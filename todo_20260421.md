   ```javascript
   /**
    * 生成数独题目
    * @param {string} difficulty - 难度级别
    * @returns {Promise<{solution, puzzle, givenMask}>}
    */
   async function generatePuzzle(difficulty) {
     const holesTarget = DIFFICULTY_HOLES[difficulty];
     const solution = generateFullBoard();
     const puzzle = digHolesWithValidation(solution, holesTarget);
     const givenMask = puzzle.map(row => row.map(cell => cell !== 0));
     return { solution, puzzle, givenMask };
   }

   /**
    * 初始化游戏UI
    * @param {number[][]} puzzle - 题目棋盘
    * @param {boolean[][]} givenMask - 预填格子标记
    */
   function initializeGameUI(puzzle, givenMask) {
     renderBoard(puzzle, givenMask);
     resetTimer();
     startTimer();
     updatePauseButton(false, i18n ? i18n.t.bind(i18n) : null);
   }

   /**
    * 处理新游戏流程（重构后）
    */
   async function handleNewGame() {
     try {
       clearGameState();
       setGlobalState('isPaused', false);

       setLoading(true);
       stopTimer();
       setTimerDisplay(i18n.t('buttons.generating'));

       await new Promise(r => setTimeout(r, 100));

       const difficulty = getDifficulty();
       const { solution, puzzle, givenMask } = await generatePuzzle(difficulty);

       gameStateManager.setGame({ solution, puzzle, givenMask });
       gameStateManager.setLastSaveTime(0);

       initializeGameUI(puzzle, givenMask);

       setLoading(false);
       showSuccess(i18n.t('puzzleGenerated', { difficulty: i18n.t(`difficulty.${difficulty}`) }));
       emit(EVENTS.GAME_STARTED, { difficulty });

       incrementGameCount();
     } catch (error) {
       console.error('生成失败:', error);
       showError(i18n.t('errors.generationFailed'));
       setLoading(false);
       emit(EVENTS.ERROR_GENERATION, { error });
     }
   }
   ```

### 7. ⏳ 修复 Supabase 配置硬编码（安全问题）
**状态**: 可选
**优先级**: 🔴 高（安全）但 🟡 中（实施复杂度）
**影响**: 安全性
**文件**:
- `public/js/config/constants.js:34-37`
- `public/js/config/env.js` (新建)
- `public/js/auth/auth-handler.js`
- `functions/_middleware.js` (新建)
**预计工作量**: 4小时

**决策点**: 需要确认是否要处理此问题

**问题分析**:
- `constants.js` 硬编码了 Supabase URL 和 anonKey
- 虽然是 anonKey（非服务端密钥），但仍然暴露在客户端
- 如果 RLS 策略配置不当，可能存在安全风险

**方案选择**:

**方案 A**: 环境变量（推荐）
- ✅ 支持多环境隔离
- ✅ 凭证不会出现在版本控制中
- ⚠️ 仍然暴露在客户端
- 实施难度：中

**方案 B**: 后端代理（最安全）
- ✅ 凭证完全不在客户端
- ✅ 最安全
- ❌ 需要额外开发 Cloudflare Functions
- 实施难度：高

**方案 C**: 仅作为配置模板
- ✅ 简单
- ✅ 适合开源项目
- ❌ 安全性提升有限
- 实施难度：低

**建议**: 先确认 Supabase 的 RLS 策略配置，再决定优先级
